165 void SetObject(int loop) // Sets The Initial Value Of Each Object (Random)

177 void LoadGLTextures() // Creates Textures From Bitmaps In The Resource File
178 {
179  HBITMAP hBMP; // Handle Of The Bitmap
180  BITMAP BMP; // Bitmap Structure
181  int loop=0;
182 
183  // The ID Of The 3 Bitmap Images We Want To Load From The Resource File
184  BYTE Texture[]={ IDB_BUTTERFLY1, IDB_BUTTERFLY2, IDB_BUTTERFLY3 };
185 
186  glGenTextures(sizeof(Texture), &texture[0]); // Generate 3 Textures (sizeof(Texture)=3 ID's)
187 
188  for (loop=0; loop<sizeof(Texture); loop++) // Loop Through All The ID's (Bitmap Images)
189  {
190  hBMP=(HBITMAP)LoadImage(GetModuleHandle(NULL),MAKEINTRESOURCE(Texture[loop]), IMAGE_BITMAP, 0, 0, LR_CREATEDIBSECTION);
191  if (hBMP) // Does The Bitmap Exist?
192  { // If So...
193  GetObject(hBMP,sizeof(BMP), &BMP); // Get The Object
194  // hBMP: Handle To Graphics Object
195  // sizeof(BMP): Size Of Buffer For Object Information
196  // Buffer For Object Information
197  glPixelStorei(GL_UNPACK_ALIGNMENT,4); // Pixel Storage Mode (Word Alignment / 4 Bytes)
198  glBindTexture(GL_TEXTURE_2D, texture[loop]); // Bind Our Texture
199  glTexParameteri(GL_TEXTURE_2D,GL_TEXTURE_MAG_FILTER,GL_LINEAR); // Linear Filtering
200  glTexParameteri(GL_TEXTURE_2D,GL_TEXTURE_MIN_FILTER,GL_LINEAR_MIPMAP_LINEAR); // Mipmap Linear Filtering
201 
202  // Generate Mipmapped Texture (3 Bytes, Width, Height And Data From The BMP)
203  gluBuild2DMipmaps(GL_TEXTURE_2D, 3, BMP.bmWidth, BMP.bmHeight, GL_BGR_EXT, GL_UNSIGNED_BYTE, BMP.bmBits);
204  DeleteObject(hBMP); // Delete The Bitmap Object
205  }
206  }

276 void DrawGL(void) // Draw The Scene
277 {
278  int loop=0;
279 
280  glClear (GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT); // Clear Screen And Depth Buffer
281 
282  for (loop=0; loop<50; loop++) // Loop Of 50 (Draw 50 Objects)
283  {
284  glLoadIdentity (); // Reset The Modelview Matrix
285  glBindTexture(GL_TEXTURE_2D, texture[obj[loop].tex]); // Bind Our Texture
286  glTranslatef(obj[loop].x,obj[loop].y,obj[loop].z); // Position The Object
287  glRotatef(45.0f,1.0f,0.0f,0.0f); // Rotate On The X-Axis
288  glRotatef((obj[loop].spinz),0.0f,0.0f,1.0f); // Spin On The Z-Axis
289 
290  glBegin(GL_TRIANGLES); // Begin Drawing Triangles
291  // First Triangle _____
292  glTexCoord2f(1.0f,1.0f); glVertex3f( 1.0f, 1.0f, 0.0f); // (2)| / (1)
293  glTexCoord2f(0.0f,1.0f); glVertex3f(-1.0f, 1.0f, obj[loop].flap); // | /
294  glTexCoord2f(0.0f,0.0f); glVertex3f(-1.0f,-1.0f, 0.0f); // (3)|/
295 
296  // Second Triangle
297  glTexCoord2f(1.0f,1.0f); glVertex3f( 1.0f, 1.0f, 0.0f); // /|(1)
298  glTexCoord2f(0.0f,0.0f); glVertex3f(-1.0f,-1.0f, 0.0f); // / |
299  glTexCoord2f(1.0f,0.0f); glVertex3f( 1.0f,-1.0f, obj[loop].flap); // (2)/____|(3)
300 
301  glEnd(); // Done Drawing Triangles
302 
303  obj[loop].y-=obj[loop].yi; // Move Object Down The Screen
304  obj[loop].spinz+=obj[loop].spinzi; // Increase Z Rotation By spinzi
305  obj[loop].flap+=obj[loop].fi; // Increase flap Value By fi
306 
307  if (obj[loop].y<-18.0f) // Is Object Off The Screen?
308  {
309  SetObject(loop); // If So, Reassign New Values
310  }
311 
312  if ((obj[loop].flap>1.0f) || (obj[loop].flap<-1.0f)) // Time To Change Flap Direction?
313  {
314  obj[loop].fi=-obj[loop].fi; // Change Direction By Making fi = -fi
315  }
316  }
317 
318  Sleep(15); // Create A Short Delay (15 Milliseconds)
319 
320  glFlush (); // Flush The GL Rendering Pipeline
321 }

510 // Process Window Message Callbacks
511 LRESULT CALLBACK WindowProc (HWND hWnd, UINT uMsg, WPARAM wParam, LPARAM lParam)
512 {
513  // Get The Window Context
514 #ifdef __DMC__ // Digital Mars Compiler don't declare GetWindowLongPtr
515  GL_Window* window = (GL_Window*)(GetWindowLong (hWnd, GWLP_USERDATA));
516 #else
517  GL_Window* window = (GL_Window*)(GetWindowLongPtr (hWnd, GWLP_USERDATA));
518 #endif
519 
520  switch (uMsg) // Evaluate Window Message
521  {
522  case WM_SYSCOMMAND: // Intercept System Commands
523  {
524  switch (wParam) // Check System Calls
525  {
526  case SC_SCREENSAVE: // Screensaver Trying To Start?
527  case SC_MONITORPOWER: // Monitor Trying To Enter Powersave?
528  return 0; // Prevent From Happening
529  }
530  break; // Exit
531  }
532 
533 
534  case WM_CREATE: // Window Creation
535  {
536  CREATESTRUCT* creation = (CREATESTRUCT*)(lParam); // Store Window Structure Pointer
537  window = (GL_Window*)(creation->lpCreateParams);
538  SetWindowLong (hWnd, GWLP_USERDATA, (LONG)(window));
539  }
540  return 0; // Return
541 
542  case WM_CLOSE: // Closing The Window
543  TerminateApplication(window); // Terminate The Application
544  return 0; // Return
545 
546  case WM_SIZE: // Size Action Has Taken Place
547  switch (wParam) // Evaluate Size Action
548  {
549  case SIZE_MINIMIZED: // Was Window Minimized?
550  window->isVisible = FALSE; // Set isVisible To False
551  return 0; // Return
552 
553  case SIZE_MAXIMIZED: // Was Window Maximized?
554  window->isVisible = TRUE; // Set isVisible To True
555  ReshapeGL (LOWORD (lParam), HIWORD (lParam)); // Reshape Window - LoWord=Width, HiWord=Height
556  return 0; // Return
557 
558  case SIZE_RESTORED: // Was Window Restored?
559  window->isVisible = TRUE; // Set isVisible To True
560  ReshapeGL (LOWORD (lParam), HIWORD (lParam)); // Reshape Window - LoWord=Width, HiWord=Height
561  return 0; // Return
562  }
563  break; // Break
564 
565  case WM_KEYDOWN: // Update Keyboard Buffers For Keys Pressed
566  if ((wParam >= 0) && (wParam <= 255)) // Is Key (wParam) In A Valid Range?
567  {
568  window->keys->keyDown [wParam] = TRUE; // Set The Selected Key (wParam) To True
569  return 0; // Return
570  }
571  break; // Break
572 
573  case WM_KEYUP: // Update Keyboard Buffers For Keys Released
574  if ((wParam >= 0) && (wParam <= 255)) // Is Key (wParam) In A Valid Range?
575  {
576  window->keys->keyDown [wParam] = FALSE; // Set The Selected Key (wParam) To False
577  return 0; // Return
578  }
579  break; // Break
580 
581  case WM_TOGGLEFULLSCREEN: // Toggle FullScreen Mode On/Off
582  g_createFullScreen = (g_createFullScreen == TRUE) ? FALSE : TRUE;
583  PostMessage (hWnd, WM_QUIT, 0, 0);
584  break; // Break
585  }
586 
587  return DefWindowProc (hWnd, uMsg, wParam, lParam); // Pass Unhandled Messages To DefWindowProc
588 }

590 BOOL RegisterWindowClass (Application* application) // Register A Window Class For This Application.
591 { // TRUE If Successful
592  // Register A Window Class
593  WNDCLASSEX windowClass; // Window Class
594  ZeroMemory (&windowClass, sizeof (WNDCLASSEX)); // Make Sure Memory Is Cleared
595  windowClass.cbSize = sizeof (WNDCLASSEX); // Size Of The windowClass Structure
596  windowClass.style = CS_HREDRAW | CS_VREDRAW | CS_OWNDC; // Redraws The Window For Any Movement / Resizing
597  windowClass.lpfnWndProc = (WNDPROC)(WindowProc); // WindowProc Handles Messages
598  windowClass.hInstance = application->hInstance; // Set The Instance
599  windowClass.hbrBackground = GetStockObject(BLACK_BRUSH); // Class Background Brush Color
600  windowClass.hCursor = LoadCursor(NULL, IDC_ARROW); // Load The Arrow Pointer
601  windowClass.lpszClassName = application->className; // Sets The Applications Classname
602  if (RegisterClassEx (&windowClass) == 0) // Did Registering The Class Fail?

163 void DrawGL(void) // Draw The Scene
164 {
165  int loop=0;
166 
167  glClear (GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT); // Clear Screen And Depth Buffer
168 
169  for (loop=0; loop<50; loop++) // Loop Of 50 (Draw 50 Objects)
170  {
171  glLoadIdentity (); // Reset The Modelview Matrix
172  glBindTexture(GL_TEXTURE_2D, texture[obj[loop].tex]); // Bind Our Texture
173  glTranslatef(obj[loop].x,obj[loop].y,obj[loop].z); // Position The Object
174  glRotatef(45.0f,1.0f,0.0f,0.0f); // Rotate On The X-Axis
175  glRotatef((obj[loop].spinz),0.0f,0.0f,1.0f); // Spin On The Z-Axis
176 
177  glBegin(GL_TRIANGLES); // Begin Drawing Triangles
178  // First Triangle _____
179  glTexCoord2f(1.0f,1.0f); glVertex3f( 1.0f, 1.0f, 0.0f); // (2)| / (1)
180  glTexCoord2f(0.0f,1.0f); glVertex3f(-1.0f, 1.0f, obj[loop].flap); // | /
181  glTexCoord2f(0.0f,0.0f); glVertex3f(-1.0f,-1.0f, 0.0f); // (3)|/
182 
183  // Second Triangle
184  glTexCoord2f(1.0f,1.0f); glVertex3f( 1.0f, 1.0f, 0.0f); // /|(1)
185  glTexCoord2f(0.0f,0.0f); glVertex3f(-1.0f,-1.0f, 0.0f); // / |
186  glTexCoord2f(1.0f,0.0f); glVertex3f( 1.0f,-1.0f, obj[loop].flap); // (2)/____|(3)
187 
188  glEnd(); // Done Drawing Triangles
189 
190  obj[loop].y-=obj[loop].yi; // Move Object Down The Screen
191  obj[loop].spinz+=obj[loop].spinzi; // Increase Z Rotation By spinzi
192  obj[loop].flap+=obj[loop].fi; // Increase flap Value By fi
193 
194  if (obj[loop].y<-18.0f) // Is Object Off The Screen?
195  {
196  SetObject(loop); // If So, Reassign New Values

198
